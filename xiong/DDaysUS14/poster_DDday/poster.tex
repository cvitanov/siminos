\headerbox{Introduction}{name=introduction,column=0,row=0,span=2}{

\begin{multicols}{2}

\textbf{\KSe\ }
	\begin{equation}
	u_t=-\frac{1}{2}(u^2)_x-u_{xx}-u_{xxxx}
	\label{eq:kse}
	\end{equation}
has been investigated intensively for its
chaotic behavior, and the periodic orbits on the attractor play an
important role in characterizing the state space.
The largest positive \textbf{Lyapunov exponent}
	\begin{equation}
		\lambda=\lim_{t\to \infty}\frac{1}{t}\ln \frac{|\delta x(t)|}{|\delta x(0)|}
	\end{equation}
has long been used as an indicator of chaotic dynamics, and recently
sets of \textbf{covariant Lyapunov vectors} have been shown to determine
the ``physical dimension'' of a strange
attractor~\cite{GiChLiPo12} of a spatially extended flow, for ex. the \KS\ on a finite periodic domain.
The evolution of the tangent space of a dynamic system is governed by the \textbf{Jacobian matrix}
	\begin{equation}
		\delta x(x_0, t)=J^t(x_0)\,\delta x(x_0, 0)
	\label{eq:jacobian}
	\end{equation}
For a periodic orbit, the eigenvalues of the Jacobian matrix are called the \textbf{Floquet multipliers}, and the associated
\textbf{Floquet eigenvectors} define the invariant directions of the tangent space.

\end{multicols}
\vspace{0.01em} % When there are two boxes, some whitespace may need to be added if the one on the right has more content

\headerbox{QR iteration}{name=qr,column=0,below=introduction}{, %bottomaligned=conclusion}{

System \eqref{eq:kse} can have a large range of Lyapunov exponents, and same is true for Floquet
multipliers of Jacobian matrix \eqref{eq:jacobian}. \textit{QR iteration}~\cite{Trefethen97}
provides a practical way to evaluate them.
	\begin{itemize}
	\item stage 1: forward transient iteration
	\item stage 2: forward dynamics
	\item stage 3: backward transient iteration
	\item stage 4: backward dynamics
	\end{itemize}
}

%	power iteration

\headerbox{Power Iteration}{name=power,column=0,span=1, below=qr,above=bottom}{ % This block is as tall as the references block
	Assume that all the eigenvalues of matrix $R$ are ordered as:
	$|\Lambda_{1}|<|\Lambda_{2}|<\cdots<|\Lambda_{N}|$. A generic initial vector:
	$x=\alpha_{1}v_{1}+\alpha_{2}v_{2}+\cdots+\alpha_{i}v_{i}$.
	\[
	R^{k}x=\Lambda_{i}^{k}(\alpha_{1}\frac{\Lambda_{1}^{k}}{\Lambda_{i}^{k}}v_{1}
			+\alpha_{2}\frac{\Lambda_{2}^{k}}{\Lambda_{i}^{k}}v_{2}
			+\cdots
			+\alpha_{i}v_{i})
	\]
	converges to the
	$i_{th}$ eigenvector of $R$:
	\[
	\lim_{k\to \infty} \frac{R^{k}x}{|R^{k}x|}=v_{i}
	\]
	In practice, power iteration and shifted power iteration are combined to
	accelerate the converging process.	
}
